# 解法
10001番目の素数は1000000以下にはありそうなので，最初にエラトステネスの篩を用いて1000000までの自然数についてそれぞれ素数判定をしておいて，for文を用いて素数の個数を数えていき，10001個目の素数に達したらその数を出力すればよい．

``` C++
#include <iostream>
#include <vector>
using namespace std;

void furui(long long N, vector<bool> &isprime) {
    isprime[0] = false;
    isprime[1] = false;
    for (long long i = 2; i * i <= N; i++) {
        if (isprime[i]) {
            for (long long j = 2; i * j <= N; j++) {
                isprime[i * j] = false;
            }
        }
    }
}

int main() {
    vector<bool> isprime(1000001, true);
    furui(1000000, isprime);
    int count = 0;
    for (int i = 0; i <= 1000000; i++) {
        if (isprime[i]) {
            count++;
        }
        if (count == 10001) {
            cout << i << endl;
            return 0;
        }
    }
}
```
